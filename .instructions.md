
- **建议行数**：80-300 行（含 JSDoc）

### `hooks/` - 自定义 React Hooks

**标准**：纯函数、完整类型、单一职责

#### 命名规范

- 文件名：`use<FunctionName>.ts`
- Hook 函数名：`use<FunctionName>`
- 例如：`useChatWebSocket.ts` 导出 `useChatWebSocket`

#### 类型要求

```typescript
// ✅ 完整的返回类型
export function useData(): {
  data: T | null
  isLoading: boolean
  error: Error | null
}

// ❌ 避免
function useData() {
  // 无返回类型注解
}
```

#### 最佳实践

```typescript
'use client'

import { useCallback, useEffect, useState } from 'react'
import type { CacheOptions } from '@/lib/types'

interface UseDataOptions {
  url: string
  cache?: CacheOptions
  onError?: (error: Error) => void
}

interface UseDataResult<T> {
  data: T | null
  isLoading: boolean
  error: Error | null
  refetch: () => Promise<void>
}

// ✅ 前置类型定义，清晰的泛型
export function useData<T = unknown>(options: UseDataOptions): UseDataResult<T> {
  const { url, cache, onError } = options
  const [data, setData] = useState<T | null>(null)
  const [isLoading, setIsLoading] = useState(false)
  const [error, setError] = useState<Error | null>(null)

  const fetchData = useCallback(async () => {
    setIsLoading(true)
    setError(null)
    try {
      const result = await fetch(url)
      if (!result.ok) throw new Error(`HTTP ${result.status}`)
      const json = await result.json()
      setData(json)
    } catch (err) {
      const error = err instanceof Error ? err : new Error(String(err))
      setError(error)
      onError?.(error)
    } finally {
      setIsLoading(false)
    }
  }, [url, onError])

  useEffect(() => {
    fetchData()
  }, [fetchData])

  const refetch = useCallback(() => fetchData(), [fetchData])

  return { data, isLoading, error, refetch }
}
```

#### 长度限制

- **推荐**: ≤ 100 行
- **含逻辑注释**: ≤ 150 行
- **超过 150 行**：考虑拆分为多个 hook

### `stores/` - Zustand 状态管理

**标准**：类型安全、Immutable、清晰的操作语义

#### 文件命名

- 文件名：`<feature>Store.ts`
- Export 名字：`use<Feature>Store`
- 例如：`authStore.ts` → `useAuthStore`

#### Store 结构标准

```typescript
import { create } from 'zustand'
import { immer } from 'zustand/middleware/immer'

// ✅ 分离状态和操作的接口
interface AuthState {
  // 状态
  user: User | null
  isLoading: boolean
  token: string | null

  // 操作
  setUser: (user: User) => void
  setToken: (token: string) => void
  logout: () => void
}

export const useAuthStore = create<AuthState>()(
  immer(set => ({
    // 初始状态
    user: null,
    isLoading: false,
    token: null,

    // 操作 - 使用 Immer 确保不变性
    setUser: (user: User) => set({ user }),
    setToken: (token: string) => set({ token }),
    logout: () =>
      set({
        user: null,
        token: null,
      }),
  }))
)

// ✅ 导出类型供使用者
export type { AuthState }
```

#### 使用规范

```typescript
// ✅ 使用选择器避免过度订阅
const user = useAuthStore(state => state.user)

// ❌ 避免订阅整个 store
const store = useAuthStore() // 每个状态变化都会重新渲染
```

#### 大型 Store 拆分

- 数据相关：`dataStore.ts`
- UI 状态：`uiStore.ts`
- 用户会话：`authStore.ts`
- 缓存：`cacheStore.ts`
- **每个 store ≤ 30 个字段**

#### 异步操作

```typescript
// ✅ 异步操作规范
interface TodoState {
  todos: Todo[]
  isLoading: boolean
  error: string | null

  fetchTodos: () => Promise<void>
}

export const useTodoStore = create<TodoState>(set => ({
  todos: [],
  isLoading: false,
  error: null,

  fetchTodos: async () => {
    set({ isLoading: true, error: null })
    try {
      const data = await fetch('/api/todos').then(r => r.json())
      set({ todos: data })
    } catch (error) {
      set({ error: error instanceof Error ? error.message : 'Unknown error' })
    } finally {
      set({ isLoading: false })
    }
  },
}))
```

#### API 客户端规范

```typescript
// ✅ 统一的 API 响应格式
export interface ApiResponse<T> {
  success: boolean
  data: T | null
  error: string | null
  code: number
}

export async function request<T = unknown>(
  endpoint: string,
  options: RequestInit = {}
): Promise<ApiResponse<T>> {
  try {
    const response = await fetch(endpoint, {
      headers: {
        'Content-Type': 'application/json',
        ...options.headers,
      },
      ...options,
    })

    if (!response.ok) {
      return {
        success: false,
        data: null,
        error: `HTTP ${response.status}`,
        code: response.status,
      }
    }

    const data = await response.json()
    return {
      success: true,
      data,
      error: null,
      code: 200,
    }
  } catch (error) {
    return {
      success: false,
      data: null,
      error: error instanceof Error ? error.message : 'Unknown error',
      code: 0,
    }
  }
}
```

#### 工具函数规范

```typescript
// ✅ 文档化、类型完整
/**
 * 格式化日期为 YYYY-MM-DD HH:mm:ss
 * @param date - 输入日期
 * @param locale - 区域设置（默认: 'zh-CN'）
 * @returns 格式化的日期字符串
 * @example
 * formatDate(new Date()) // '2026-03-06 14:30:00'
 */
export function formatDate(date: Date | string, locale = 'zh-CN'): string {
  const dateObj = typeof date === 'string' ? new Date(date) : date
  if (isNaN(dateObj.getTime())) {
    throw new Error(`Invalid date: ${date}`)
  }

  return dateObj.toLocaleString(locale, {
    year: 'numeric',
    month: '2-digit',
    day: '2-digit',
    hour: '2-digit',
    minute: '2-digit',
    second: '2-digit',
  })
}
```

## ✅ 通用规范

### TypeScript 检查清单

- [ ] 没有 `any` 类型使用（除非带 `@ts-expect-error` 注释）
- [ ] 所有函数有返回值类型
- [ ] Props/参数接口完整
- [ ] 泛型约束清晰（`<T extends Base>`）

### 样式检查清单

- [ ] 使用 Tailwind class（不用内联 style）
- [ ] 颜色用 theme 变量（`bg-primary`）
- [ ] 间距规范（`gap-4`、`p-4`）
- [ ] 响应式前缀正确（`md:`、`lg:`）

### 代码质量检查清单

- [ ] 函数 ≤ 50 行
- [ ] 嵌套深度 ≤ 4
- [ ] 变量命名清晰
- [ ] 复杂逻辑有注释
- [ ] 没有控制台警告
